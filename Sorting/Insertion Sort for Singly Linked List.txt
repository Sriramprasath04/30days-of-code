class Solution
{
    public static Node insertionSort(Node head)
    {
        Node dummy = new Node(0);
        dummy.next = head;
        head = dummy;
        Node curr = head.next;
        Node prev = head;
        while(curr!=null){
            if(curr.data>=prev.data){
                prev = curr;
                curr = curr.next;
                continue;
            }
            
            Node temp = head;
            while(curr.data > temp.next.data){
                temp = temp.next;
            }
            prev.next = curr.next;
            curr.next = temp.next;
            temp.next = curr;
            curr = prev.next;
        }
        
        head = dummy.next;
        return head;
    }
}
